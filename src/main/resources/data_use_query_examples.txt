# Data use query examples

# Example 1: Simple match query
POST /content/_search
{
  "query": {
    "match": { "search_text": "home" }
  }
}

# Example 2: Boolean query
POST /content/_search
{
  "query": {
    "bool": {
      "must": [
        { "match": { "title": "biodiversity" } },
        { "term": { "contentType": "dataUse" } }
      ]
    }
  }
}
